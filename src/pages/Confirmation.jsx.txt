import { useEffect, useState } from 'react'
import { useLocation, useNavigate } from 'react-router'
import { sendTelegramNotification } from '../utils/notifications'
import { saveEmergency } from '../utils/firebase'


export default function Confirmation() {
  const { state } = useLocation()
  const navigate = useNavigate()
  const [isSending, setIsSending] = useState(false)
  const [notificationSent, setNotificationSent] = useState(false)
  const [error, setError] = useState(null)

  useEffect(() => {
    if (!state?.selectedHospital) {
      navigate('/hospitals')
    }
  }, [state, navigate])

  const handleConfirm = async () => {
    setIsSending(true)
    setError(null)
    
    try {
      // Save to Firebase
      const emergencyId = await saveEmergency({
        patient: state.patientDetails,
        emergencyType: state.emergencyType,
        location: state.position,
        hospital: state.selectedHospital,
        timestamp: new Date().toISOString()
      })

      // Send Telegram notification
      await sendTelegramNotification({
        hospital: state.selectedHospital,
        patient: state.patientDetails,
        emergencyType: state.emergencyType,
        eta: Math.round(state.selectedHospital.route.duration / 60)
      })

      setNotificationSent(true)
    } catch (err) {
      console.error('Notification error:', err)
      setError('Failed to send notifications. Please try again.')
    } finally {
      setIsSending(false)
    }
  }

  if (!state?.selectedHospital) return null

  return (
    <div className="min-h-screen bg-gray-100 p-6">
      <div className="max-w-2xl mx-auto bg-white rounded-lg shadow-md overflow-hidden">
        <div className="bg-blue-600 p-4 text-white">
          <h1 className="text-2xl font-bold">Emergency Confirmed</h1>
        </div>

        <div className="p-6 space-y-6">
          {/* Patient Summary */}
          <div className="border-b pb-4">
            <h2 className="text-xl font-semibold mb-2">Patient Information</h2>
            <div className="grid grid-cols-2 gap-4">
              <div>
                <p className="text-gray-600">Name</p>
                <p className="font-medium">{state.patientDetails.name || 'Not provided'}</p>
              </div>
              <div>
                <p className="text-gray-600">Age</p>
                <p className="font-medium">{state.patientDetails.age || 'Not provided'}</p>
              </div>
              <div className="col-span-2">
                <p className="text-gray-600">Condition</p>
                <p className="font-medium">{state.patientDetails.condition || 'Not provided'}</p>
              </div>
            </div>
          </div>

          {/* Hospital Summary */}
          <div className="border-b pb-4">
            <h2 className="text-xl font-semibold mb-2">Hospital Details</h2>
            <div className="space-y-2">
              <p className="font-medium">{state.selectedHospital.name}</p>
              {state.selectedHospital.address && (
                <p className="text-gray-600">{state.selectedHospital.address}</p>
              )}
              {state.selectedHospital.specialty && (
                <p className="text-blue-600 font-medium">
                  Specialty: {state.selectedHospital.specialty}
                </p>
              )}
              <p className="text-gray-600">
                Estimated arrival: {Math.round(state.selectedHospital.route.duration / 60)} minutes
              </p>
            </div>
          </div>

          {/* Notification Status */}
          <div className="bg-blue-50 p-4 rounded-lg">
            {notificationSent ? (
              <div className="text-green-600 font-medium">
                ✓ Hospital notified and preparing for arrival
              </div>
            ) : error ? (
              <div className="text-red-600">{error}</div>
            ) : (
              <div className="text-blue-600">
                {isSending ? 'Sending notifications...' : 'Ready to notify hospital'}
              </div>
            )}
          </div>

          {/* Actions */}
          <div className="flex justify-between pt-4">
            <button
              onClick={() => navigate('/hospitals', { state })}
              className="px-4 py-2 border border-gray-300 rounded-md text-gray-700 hover:bg-gray-50"
            >
              ← Back
            </button>
            <button
              onClick={handleConfirm}
              disabled={isSending || notificationSent}
              className={`px-4 py-2 rounded-md text-white ${notificationSent 
                ? 'bg-green-600' 
                : isSending 
                  ? 'bg-blue-400' 
                  : 'bg-blue-600 hover:bg-blue-700'
              }`}
            >
              {notificationSent ? 'Notifications Sent' : 'Confirm & Notify Hospital'}
            </button>
          </div>
        </div>
      </div>
    </div>
  )
}
